각 성별 신뢰구간 95% 구하기

from scipy.stats import norm
z = norm.ppf(0.975,loc=0,scale=1)

male_0975 = male_mean + z * male_scale / np.sqrt(male_n)
male_0025 = male_mean - z * male_scale / np.sqrt(male_n)

female_0975 = female_mean + z * female_scale / np.sqrt(female_n)
female_0025 = female_mean - z * female_scale / np.sqrt(female_n)

print("남성의 신뢰구간은 (", male_0025, ",", male_0975, ")")
print("여성의 신뢰구간은 (", female_0025, ",", female_0975, ")")
